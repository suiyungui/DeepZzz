
```ini

Environment=HOST=0.0.0.0
Environment=PORT=7860
Environment=ENABLE_POSE=1
Environment=POSE_MODEL=/home/z/.brdk_models/pose/yolov8n-pose-320.onnx
Environment=POSE_PROVIDER=CPUExecutionProvider
Environment=POSE_PROVIDER_FALLBACK=0
Environment=ANALYSIS_WIDTH=320
Environment=ANALYSIS_FPS=3
Environment=VIDEO_DECODER=h264_stcodec
Environment=AUDIO_DEVICE=plughw:CARD=Device,DEV=0
Environment=AUDIO_RATE=16000
Environment=YAMNET_SECONDS=2
Environment=YAMNET_INTERVAL=0.5
Environment=HLS_TIME=1
Environment=HLS_LIST_SIZE=2
ExecStart=/usr/bin/python3 /home/z/main.py
StandardOutput=append:/home/z/runtime/logs/app.out.log
StandardError=append:/home/z/runtime/logs/app.err.log
Restart=always
```

## 迁移到新板子的要点

1. 把顶层文件和目录复制到新板 `/home/z`。
2. 安装依赖：参考 `requirements-board.txt`。
3. 安装自启服务：

```bash
sudo cp /home/z/scripts/deepzzz-k2-edge.service /etc/systemd/system/deepzzz-k2-edge.service
sudo cp /home/z/scripts/day-night-sync.service /etc/systemd/system/day-night-sync.service
sudo systemctl daemon-reload
sudo systemctl enable --now deepzzz-k2-edge.service
sudo systemctl enable --now day-night-sync.service
```

4. 旧摄像头服务不要同时启用：

```bash
sudo systemctl disable --now camera-hls.service camera-web.service 2>/dev/null || true
```

5. 旧源码目录不再保留：

```text
/home/z/deepzzz_k2_edge
/home/z/day_night
/home/z/ir-cut
/home/z/camera_web
/home/z/camera-web
```
